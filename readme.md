git config --global user.name "Name"	
--> Set your Git username

git config --global user.email "email"	
--> Set your Git email

git config --list	
--> View all Git configurations

git init
--> Initialize a new Git repository

git clone <url>
--> Clone a remote repository

git commit -m "message"
--> Commit staged changes

git status	
--> Show working tree status

git add <file>	
--> Stage a file

git add .	
--> Stage all changes

git branch	
--> List branches

git branch <name>	
--> Create a new branch

git checkout <branch>
--> Switch branches

Remote Repositories

git push -u origin <branch>
--> Set upstream branch

git push
--> Push changes
